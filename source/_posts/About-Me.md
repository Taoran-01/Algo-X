### 码风
一直用快读，不用`cin`和`cout`。  
不喜欢用万能头  
数组的名字叫`d[]`，`g[]`和`f[]`，从1开始。  
不常用`const int`，一般是`#define`  。

```
#include <cmath>
#include <ctime>

#include <algorithm>
using namespace std;
typedef long long int ll;

inline ll read() {
	ll x=0, f=1;
	char ch=getchar();
	while (ch<'0'||ch>'9') {if (ch=='-') f=-1;ch=getchar();}
	while (ch>='0'&&ch<='9') {x=(x<<3)+(x<<1)+(ch^48);ch=getchar();}
	return x*f;
}
```
